85 /* Slightly smaller way to check if tag is a string tag

87 #define MBEDTLS_ASN1_IS_STRING_TAG(tag) \
88  ((unsigned int) (tag) < 32u && ( \
89  ((1u << (tag)) & ((1u << MBEDTLS_ASN1_BMP_STRING) | \
90  (1u << MBEDTLS_ASN1_UTF8_STRING) | \
